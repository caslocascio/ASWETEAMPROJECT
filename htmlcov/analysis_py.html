37def review_sentiment_analysis(df, pol_threshold=0.1, subj_threshold=0.5): 

38 """ 

39 Score and lable review's sentiment

41 # text pre-processing 

42 # lower case 

43 df['review'] = df['review'].str.lower() 

44 # remove punctuation

46 '', regex=True) 

47 # remove stopwords 

48 stop = stopwords.words('english') 

49 df['review'] = df['review'].apply(lambda x: ' '.join( 

50 [word for word in str(x).split() 

51 if word not in (stop)])) 

52 # word stemming 

53 st = PorterStemmer() 

54 df['review'] = df['review'].apply(lambda x: ' '. 

55 join([st.stem(word) for word 

56 in str(x).split()])) 

57 # TextBlob returns polarity & subjectivity scores 

58 df['senti_score'] = df['review'].apply(lambda x: TextBlob(x).sentiment) 

59 df['polarity_score'] = df['senti_score'].apply(lambda x: x[0]) 

60 df['subjectivity_score'] = df['senti_score'].apply(lambda x: x[1]) 

61 # decide if the review is positive or negative based on the threshold 

62 df['senti_fl'] = 'neutral' 

63 df.loc[df['polarity_score'] < -pol_threshold, 'senti_fl'] = 'negative' 

64 df.loc[df['polarity_score'] > pol_threshold, 'senti_fl'] = 'positive' 

65 # decide if the review is objective or subjective based on the threshold 

66 df['sub_fl'] = 'objective' 

67 df.loc[df['subjectivity_score'] > 

68 subj_threshold, 'sub_fl'] = 'subjective' 

69 # return counts 

70 return df, df[df['senti_fl'] == 'positive'].shape[0],\ 

71 df[df['senti_fl'] == 'neutral'].shape[0],\
